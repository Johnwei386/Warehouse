\subsection{多行公式}
%无需对齐的长公式可以使用multline环境
\begin{multline*}
x = a+b+c+{}\\
d+e+f+g
\end{multline*}

%需要对齐的长公式，可以使用aligned次环境来实现，它必须包含在数学环境之内
\[
\begin{aligned}
x={}&a+b+c+{}\\
&d+e+f+g
\end{aligned}
\]

%无需对齐的公式组可以使用gather环境，需要对齐的公式组可以使用align环境。加上*可以去除编号
\begin{gather}
a=b+c+d\\
x=y+z
\end{gather}
\begin{align}
a &= b+c+d\\
x &= y+z
\end{align}
